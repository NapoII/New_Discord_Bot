def delt_str_time(sec_to_delta):
    """
    ~~This mgs gets delt in 60 sec~~"

    """
    current_unix_time = int(time.time())
    delt_unix_time = discord_time_convert(current_unix_time + sec_to_delta)
    delt_msg_str = f"~~This mgs gets delt {delt_unix_time}~~"
    return delt_msg_str


def discord_time_convert(time, type=False):
    """
    `<t:1708536408:t>` = `t` = 18:26 
    `<t:1708536421:T>` = `T` = 18:27:01 
    `<t:1708536431:d>` = `d` = 21/02/2024 
    `<t:1708536440:D>` = `D` = 21 February 2024 
    `<t:1708536455:f>` = `f` = 21 February 2024 18:27 
    `<t:1708536477:R>` = `R` = a minute ago
    """
    time = int(str(time)[:10])

    if type:
        discord_time = (f"<t:{time}:{type}>")
        return discord_time
    else:
        discord_time = (f"<t:{time}:R>")
        return discord_time

    """
